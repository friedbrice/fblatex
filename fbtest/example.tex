  \begin{problem}{1}
    A GRE-style problem where we give them three statements and ask them
    which ones are true. How mean!

    Consider the following statements:

    \begin{statements}
      \statement{} $f$ is continuous at $a$.\llabel{s:1}
      \statement{} $f$ is differentiable at $a$.\llabel{s:2}
      \statement{} $f$ is integrable at $a$.\llabel{s:3}
    \end{statements}

    Which of the above statements are true?

    \begin{choices}
      \choice{} Statement \lref{s:1} only
      \choice{} Statement \lref{s:2} only
      \choice{} Statements \lref{s:1} and \lref{s:3}, but not \lref{s:2}
      \choice{} Statements \lref{s:2} and \lref{s:3}, but not \lref{s:1}
        \llabel{c:ans}
      \choice{} None of the above
    \end{choices}

    \solution{
      Clearly, it's statements \lref{s:2} and \lref{s:3}! That
      would be anwer choice \lref{c:ans}.
    }
  \end{problem}
